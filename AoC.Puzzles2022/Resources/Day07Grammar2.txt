
#DEFINITIONS

FileName        = "a file name" | [_A-Za-z][_A-Za-z0-9]*(\.)?[_A-Za-z][_A-Za-z0-9]*
Number          = "a number" | (\+|-)?[0-9]+

#GRAMMAR

line            = "$" command
                | "dir" FileName c_dir
                | Number FileName c_file

command         = "cd" directory
                | "ls" c_ls

directory       = "/" c_cdRoot
                | "." "." c_cdParent
                | FileName c_cdChild

#END
